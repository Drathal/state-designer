# whenIn

The `useStateDesign` hook returns a helper naemd `whenIn` that you can use to compose values based on the state's currently active states.

```jsx
import React from "react"
import { useStateDesign } from "state-designer"

export default function Example() {
  const { send, whenIn } = useStateDesign({
    initial: "cool",
    states: {
      cold: {},
      cool: {},
      warm: {},
      hot: {},
    },
  })

  return (
    <div>
      <h2>Count: {data.count}</h2>
      <button onClick={() => send("TOGGLED")}>Toggle</button>
      <button disabled={!isIn("active")} onClick={() => send("PLUS_CLICKED")}>
        +
      </button>
      <button disabled={!isIn("active")} onClick={() => send("MINUS_CLICKED")}>
        -
      </button>
    </div>
  )
}
```
